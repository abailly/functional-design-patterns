015    package com.google.common.hash;
016    
017    import com.google.common.annotations.Beta;
018    
019    import java.io.Serializable;
020    
021    /**
022     * An object which can send data from an object of type {@code T} into a {@code PrimitiveSink}.
023     * 
024     * @author Dimitris Andreou
025     * @since 11.0
026     */
027    @Beta
028    public interface Funnel<T> extends Serializable {
029    
030      /**
031       * Sends a stream of data from the {@code from} object into the sink {@code into}. There
032       * is no requirement that this data be complete enough to fully reconstitute the object
033       * later.
034       *
035       * @since 12.0 (in 11.0 version, {@code PrimitiveSink} was still called {@code Sink})
036       */
037      void funnel(T from, PrimitiveSink into);
038    }
